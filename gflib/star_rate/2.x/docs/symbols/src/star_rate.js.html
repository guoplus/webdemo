 15 /**

 27  * @see 参考 star_rate.init-ins
 28  */
 29 star_rate.init = function(sWrapId,aItem){

 31     /**
 32      * 克隆对象
 33      *
 34      * @param obj
 35      * @return {*}
 36      */
 37     function clone(obj) {
 38         var o;
 39         if (typeof obj == "object") {
 40             if (obj === null) {
 41                 o = null;
 42             } else {
 43                 if (obj instanceof Array) {
 44                     o = [];
 45                     for (var i = 0, len = obj.length; i < len; i++) {
 46                         o.push(clone(obj[i]));
 47                     }
 48                 } else {
 49                     o = {};
 50                     for (var j in obj) {
 51                         o[j] = clone(obj[j]);
 52                     }
 53                 }
 54             }
 55         } else {
 56             o = obj;
 57         }
 58         return o;
 59     }

 61     /**
 62      * star_rate.init返回的打分对象。
 63      *
 64      * @constructor
 65      */
 66     var ins = {
 67         /**
 68          * dom结构。
 69          *
 70          * @type Array
 71          * @field
 72          * @example
 73          * var d = document.getElementById('_id');
 74          * aWrap.push(d);
 75          */
 76         aWrap : null,
 77         /**
 78          * 打分数据。
 79          *
 80          * @type Array
 81          * @field
 82          * @example
 83          * var s = {id:"id1",title:"服务态度",tip:"很不满意::不满意::一般::满意::非常满意",score:1};
 84          * aItem.push(s);
 85          */
 86         aItem : null,
 87         /**
 88          * 回调函数。
 89          *
 90          * @type Object
 91          * @field
 92          * @example
 93          * var f = function(){}
 94          * aCallback['set'] = f;
 95          */
 96         oCallback : null,
 97 
 98         /**
 99          * 属性。
100          *
101          * @type Object
102          * @field
103          * @example
104          * oAttr['disable'] = false;
105          */
106         oAttr : {
107             disable : false
108         },
109 
110         /**
111          * 返回dom对象
112          *
113          * @param {String|Object} soD dom元素id或者dom元素对象
114          * @private
115          * @function
116          * @returns {Object} dom元素对象
117          * @example
118          * var d = ins._$("d");
119          */
120         _$ : function(soD){
121             if(!soD){
122                 return null;
123             }
124             var dom = null;
125             if(typeof soD == 'string'){
126                 dom = document.getElementById(soD);
127             }
128             else{
129                 dom = soD;
130             }
131 
132             return dom;
133         },

135         /**
136          * 生成星标打分dom对象
137          *
138          * @param {Array} aItem 打分数据
139          * @private
140          * @function
141          * @returns {Object} 星标打分dom对象
142          * @see 参数示例见 star_rate.init-ins._init
143          */
144         _genDom : function(aItem){
145             if(!aItem){
146                 return null;
147             }
148 
149             this.aItem = [];
150             this.aWrap = [];
151 
152             var df = document.createDocumentFragment();
153 
154             for(var i= 0,len=aItem.length;i<len;i++){
155                 var oc = clone(aItem[i]);
156                 this.aItem.push(oc);
157                 //
158                 var d = document.createElement('div');
159                 d.className = "star_rate";
160                 d.innerHTML = "<span class=\"title\">"+aItem[i]['title']+"</span>"
161                     +"<ul>"
162                     +"    <li><a href=\"javascript:;\">1</a></li>"
163                     +"    <li><a href=\"javascript:;\">2</a></li>"
164                     +"    <li><a href=\"javascript:;\">3</a></li>"
165                     +"    <li><a href=\"javascript:;\">4</a></li>"
166                     +"    <li><a href=\"javascript:;\">5</a></li>"
167                     +"</ul>"
168                     +"<span class=\"tip\"></span>"
169                     +"<p class=\"tip_h\"></p>";
170                 df.appendChild(d);
171                 this.aWrap.push(d);
172             }
173 
174             return df;
175         },

177         /**
178          * 初始化打分组件。
179          *
180          * @private
181          * @function
182          * @param {String} sWrapId 放置打分内容的容器id。
183          * @param {Array} aItem 打分数据。
184          * @see 参数示例见 star_rate.init
185          */
186         _init : function(sWrapId,aItem){
187             if(!sWrapId || !aItem){
188                 return;
189             }
190             var wrapDom = this._$(sWrapId);
191             if(!wrapDom){
192                 return;
193             }
194 
195             // 生成文档结构
196             var doms = this._genDom(aItem);
197             if(!doms){
198                 return;
199             }
200             wrapDom.appendChild(doms);
201 
202             //
203             var _ins = this;
204 
205             // 设置事件响应
206             for(var wi = 0,wlen=_ins.aWrap.length;wi<wlen;wi++){
207                 // 获取dom
208                 var oDiv = _ins.aWrap[wi];
209                 var oUl = oDiv.getElementsByTagName("ul")[0];
210                 var oP = oDiv.getElementsByTagName("p")[0]; // 悬浮提示
211                 var aLi = oUl.getElementsByTagName("li"); // 星星
212                 // 数据
213                 var oItem = _ins.aItem[wi];
214                 // 提示信息
215                 var sTip = oItem['tip'];
216                 var aTip = null;
217                 if(sTip){
218                     aTip = sTip.split('::');
219                     if(aTip.length!=5){
220                         aTip = null;
221                     }
222                 }
223                 // 其他数据
224                 var iScore = oItem['score'];
225                 if(!iScore){
226                     iScore = 0;
227                 }
228                 var sId = oItem['id'];
229 
230                 // 添加事件
231                 for (var i = 0; i < aLi.length; i++){
232                     aLi[i].score = i+1;
233                     aLi[i].sId = sId;
234                     aLi[i].oP = oP;
235                     aLi[i].oUl = oUl;
236 
237                     // 鼠标移过显示分数
238                     aLi[i].onmouseover = function (){
239                         //
240                         if(!_ins._acl('mouseover')){
241                             return;
242                         }
243 
244                         // 显示
245                         var oS = {id:this.sId,score:this.score};
246                         var aS = [oS];
247                         _ins.show(aS);
248                         // 显示浮层
249                         // 计算浮动层位置
250                         this.oP.style.left = this.oUl.offsetLeft + this.score * this.offsetWidth + "px";
251                         this.oP.style.display = "block";
252                     };
253 
254                     // 鼠标离开后恢复上次评分
255                     aLi[i].onmouseout = function (){
256                         //
257                         if(!_ins._acl('mouseout')){
258                             return;
259                         }
260 
261                         // 显示
262                         _ins.show();
263                         // 隐藏浮动层
264                         this.oP.style.display = "none";
265                     };
266 
267                     // 点击后进行评分处理
268                     aLi[i].onclick = function (){
269                         //
270                         if(!_ins._acl('click')){
271                             return;
272                         }
273 
274                         // 设置
275                         var oS = {id:this.sId,score:this.score};
276                         var aS = [oS];
277                         _ins.set(aS);
278                         // 显示
279                         _ins.show();
280                         // 隐藏浮动层
281                         this.oP.style.display = "none";
282                     };
283                 }
284             }
285         },

287         /**
288          * 设置分数。
289          *
290          * @function
291          * @param {Array} aItem
292          * @example
293          * var s = [{id:"id1",score:3},{id:"id2",score:4}];
294          * ins.set(s);
295          */
296         set : function(aItem){
297             var _ins = this;
298 
299             // 回调
300             if(_ins.oCallback && _ins.oCallback['set_before']){
301                 _ins.oCallback['set_before'](_ins,aItem);
302             }
303 
304             if(!aItem){
305                 return;
306             }
307 
308             var aItemOri = _ins.aItem;
309             if(!aItemOri){
310                 return;
311             }
312 
313             for(var ai = 0,alen = aItem.length;ai<alen;ai++){
314                 var oItem = aItem[ai];
315                 var s = oItem['score'];
316                 if(!s){
317                     s = 0;
318                 }
319                 for(var aj = 0,ajlen = aItemOri.length;aj<ajlen;aj++){
320                     var oItemOri = aItemOri[aj];
321                     if(oItem['id']==oItemOri['id']){
322                         oItemOri['score'] = s;
323                         break;
324                     }
325                 }
326             }
327 
328             _ins.show(aItem);
329 
330             // 回调
331             if(_ins.oCallback && _ins.oCallback['set_after']){
332                 _ins.oCallback['set_after'](_ins,aItem);
333             }
334         },
335 
336         /**
337          * 显示分数。
338          *
339          * @function
340          * @param {Array|null} aItem
341          * @example
342          * var s = [{id:"id1"},{id:"id2"}];
343          * ins.show(s);
344          * 或者
345          * var s = [{id:"id1",score:3},{id:"id2",score:4}];
346          * ins.show(s);
347          * 或者
348          * ins.show();
349          */
350         show : function(aItem){
351             var _ins = this;
352             if(!aItem){
353                 aItem = _ins.aItem;
354             }
355 
356             var aItemOri = _ins.aItem;
357 
358             for(var ai = 0,alen = aItem.length;ai<alen;ai++){
359                 var oItem = aItem[ai];
360                 var wi = -1; // dom索引
361                 // 获得dom序号
362                 for(var aj = 0,ajlen = aItemOri.length;aj<ajlen;aj++){
363                     if(oItem['id']==aItemOri[aj]['id']){
364                         wi = aj;
365                         break;
366                     }
367                 }
368                 if(wi < 0){
369                     continue;
370                 }
371                 //
372                 // 获取dom
373                 var oDiv = _ins.aWrap[wi];
374                 var oUl = oDiv.getElementsByTagName("ul")[0];
375                 var oSpan = oDiv.getElementsByTagName("span")[1]; // 提示
376                 var oP = oDiv.getElementsByTagName("p")[0]; // 悬浮提示
377                 var aLi = oUl.getElementsByTagName("li"); // 星星
378                 // 数据
379                 var oItemOri = aItemOri[wi];
380                 // 提示信息
381                 var sTip = oItemOri['tip'];
382                 var aTip = null;
383                 if(sTip){
384                     aTip = sTip.split('::');
385                     if(aTip.length!=5){
386                         aTip = null;
387                     }
388                 }
389                 // 其他数据
390                 var iScore = oItem['score'];
391                 if(!iScore){
392                     iScore = oItemOri['score'];
393                 }
394                 if(!iScore){
395                     iScore = 0;
396                 }
397 
398                 // 星星
399                 var cln = 'on';
400                 if(iScore<3){
401                     cln = 'sad';
402                 }
403                 for (var si = 0; si < aLi.length; si++){
404                     if(si < iScore){
405                         aLi[si].className = cln;
406                     }
407                     else{
408                         aLi[si].className = "";
409                     }
410                 }
411 
412                 // 浮动层
413                 var s = '';
414                 if(aTip) {
415                     s = aTip[iScore-1];
416                 }
417                 oP.innerHTML = "<em><b>" + iScore + "</b> 分 " + s + "</em>";
418 
419                 // 提示
420                 if(iScore>0) {
421                     oSpan.innerHTML = "<strong>" + (iScore) + " 分</strong> " + s;
422 
423                 } else {
424                     oSpan.innerHTML = "";
425                 }
426             }
427 
428         },
429 
430         /**
431          * 设置回调函数。
432          *
433          * @function
434          * @param {Object} oFunc
435          * @example
436          * function func(ins,aItem){}
437          * var cb = {set_after:func};
438          * ins.setCallback(cb);
439          *
440          * 允许的参数值如下：
441          * key :: 回调 参数
442          * ================
443          * set_after :: ins实例,set参数
444          * set_before :: ins实例,set参数
445          */
446         setCallback : function(oFunc){
447             if(!oFunc){
448                 return;
449             }
450 
451             this.oCallback = this.oCallback || {};
452 
453             for (var v in oFunc) {
454                 this.oCallback[v] = oFunc[v];
455             }
456 
457         },

459         /**
460          * 设置属性。
461          *
462          * @function
463          * @param {Object} oAttr
464          * @example
465          *
466          * var a = {disable:false};
467          * ins.setAttr(a);
468          *
469          * 允许的参数值如下：
470          * key :: 描述 :: value 可选值 :: 默认值
471          * ===================================
472          * disable :: 是否允许打分 :: true或false :: false
473          *
474          */
475         setAttr : function(oAttr){
476             if(!oAttr){
477                 return;
478             }
479 
480             for (var v in oAttr) {
481                 this.oAttr[v] = oAttr[v];
482             }
483         },
484 
485         /**
486          * 判断是否可操作。
487          *
488          * @function
489          * @private
490          * @param {String} sAct
491          * @returns {Boolean} 默认false
492          * @example
493          *
494          * if(acl('mouseover')){
495          *     // ...
496          * }
497          *
498          * 可选参数如下：
499          * mouseover
500          * click
501          * mouseout
502          */
503         _acl : function(sAct){
504             var re = false;
505 
506             if(sAct=='mouseover') {
507                 if(!this.oAttr['disable']){
508                     re = true;
509                 }
510             } else if(sAct=='click') {
511                 if(!this.oAttr['disable']){
512                     re = true;
513                 }
514             } else if (sAct=='mouseout') {
515                 if(!this.oAttr['disable']){
516                     re = true;
517                 }
518             }
519 
520             return re;
521         }
522     };
523 
524     ins._init(sWrapId, aItem);
525 
526     return ins;
527 };
